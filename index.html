        // Reminder System

        function checkReminders() {

            const now = new Date();

            const currentTime = `${String(now.getHours()).padStart(2, '0')}:${String(now.getMinutes()).padStart(2, '0')}`;

            

            tasks.forEach(task => {

                if (shouldTriggerReminder(task, currentTime)) {

                    triggerReminder(task.id);

                    task.lastNotified = now.toISOString();

                    saveTasks();

                }

            });

        }

        

        function shouldTriggerReminder(task, currentTime) {

            // Check if already notified recently (within 30 minutes)

            if (task.lastNotified) {

                const lastNotified = new Date(task.lastNotified);

                const now = new Date();

                const diffMinutes = (now - lastNotified) / 60000;

                if (diffMinutes < 30) return false;

            }

            

            // Check if already completed today

            if (isTaskCompletedToday(task)) return false;

            

            // Check frequency

            const today = new Date().getDay();

            if (task.frequency === 'weekdays' && (today === 0 || today === 6)) return false;

            if (task.frequency === 'weekends' && today > 0 && today < 6) return false;

            if (task.frequency === 'once' && task.completions.length > 0) return false;

            

            // Check time with randomization

            let targetTime = task.time;

            if (task.randomization) {

                const [hours, minutes] = targetTime.split(':').map(Number);

                const totalMinutes = hours * 60 + minutes;

                const randomOffset = Math.floor(Math.random() * 21) - 10; // -10 to +10

                const adjustedMinutes = totalMinutes + randomOffset;

                const adjustedHours = Math.floor(adjustedMinutes / 60);

                const adjustedMins = adjustedMinutes % 60;

                targetTime = `${String(adjustedHours).padStart(2, '0')}:${String(adjustedMins).padStart(2, '0')}`;

            }

            

            return currentTime === targetTime;

        }

        async function completeTask() {

            if (!currentTask) return;

            

            if (currentTask.proofRequired === 'strict') {

                const photoInput = document.getElementById('photoInput');

                if (!photoInput.files.length) {

                    alert('Please upload a photo as proof!');

                    return;

                }

            }

            

            // Mark as completed

            currentTask.completions.push(new Date().toISOString());

            

            // Update streak

            const yesterday = new Date();

            yesterday.setDate(yesterday.getDate() - 1);

            const completedYesterday = currentTask.completions.some(date => 

                new Date(date).toDateString() === yesterday.toDateString()

            );

            currentTask.streak = completedYesterday ? currentTask.streak + 1 : 1;

            

            // Update stats

            stats.completed++;

            stats.streak = Math.max(stats.streak, currentTask.streak);

            saveStats();

            

            saveTasks();

            updateTasksList();

            updateStats();

            

            // Get AI congratulations

            const aiResponse = await getAIResponse(`User completed task: ${currentTask.name}. Congratulate them! Streak: ${currentTask.streak} days.`);

            showNotification('Task Completed! 🎉', aiResponse);

            

            closeModal();

        }

            }

        });
